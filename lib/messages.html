toHexLiterals()
¶

Returns an array of hex literal strings:

['0x33', '0xEF', '0xF9', ...]

toHexLiterals = (str) ->                                                                                               
  str.match(/\w{2}/g).map (byte) ->                                                                                 
    "0x#{byte}"

toBuffer = (str) ->
  new Buffer(toHexLiterals str)

messages[k]=toBuffer(v) for k, v of messages

module.exports = messages
